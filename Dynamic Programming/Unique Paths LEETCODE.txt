class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        grid = [[1 for i in range(m)] for j in range(n)]
        grid[n-1][m-1] = 1
        row = n
        col = m
        
        for i in range(1,row):
            for j in range(1,col):
                grid[i][j] = grid[i-1][j] + grid[i][j-1]
                
        return grid[-1][-1]
